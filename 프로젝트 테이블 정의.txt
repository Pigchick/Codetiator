create table user_list ( --<유저>

	user_num NUMBER PRIMARY KEY,       -- 유저번호

        chk_receive_email VARCHAR2(1), --이메일 수신여부
        jumin VARCHAR2(20), --주민등록번호

        addr VARCHAR2(80), --주소
        grade VARCHAR2(10), --등급
        join_date DATE, --가입일
	phone_key VARCHAR2(1), --휴대폰 인증
	login_time DATE --접속이력	
);

CREATE TABLE account( --<계좌>
	account_num NUMBER PRIMARY KEY, --계좌순서
	account_name VARCHAR2(10), --예금주
	bank_name VARCHAR2(10) NOT NULL, --은행명
	bank_num VARCHAR2(20) NOT NULL, --계좌번호
	deposit VARCHAR2(30), --예치금
	user_num NUMBER REFERENCES user_list(user_num) on delete cascade, --유저 번호
	busi_num NUMBER REFERENCES busi_user_list(busi_num) on delete cascade --법인 유저 번호
);

CREATE TABLE    project_list ( --<프로젝트>
        project_num NUMBER PRIMARY KEY, --프로젝트 번호
	user_num NUMBER REFERENCES user_list(user_num) on delete cascade, --유저 번호
        img VARCHAR2(60), --사진(경로)
        project_name VARCHAR2(60), --이름
	current_price VARCHAR2(30), --현재금액
        price VARCHAR2(30), --목표금액
	deadline VARCHAR2(30), --남은기간
	ach_rate VARCHAR2(10), --달성률
	customer VARCHAR2(20), --투자자수
	type VARCHAR2(10), --게임유형
	genre VARCHAR2(10), --장르
        more_information VARCHAR2(2000), --상세정보
	en_date DATE, --등록날짜
	busi_num NUMBER REFERENCES busi_user_list(busi_num) on delete cascade, --법인 유저 번호
	rate VARCHAR2(10) --금리
);

CREATE TABLE guarantee ( --<담보>
	guarantee_num NUMBER PRIMARY KEY, --담보순서
	guarantee_type VARCHAR2(20) NOT NULL, --담보유형
	guarantee_name VARCHAR2(30) NOT NULL, --담보이름
	connoisseur VARCHAR2(30), --감정가
	project_num NUMBER REFERENCES project_list(project_num) on delete cascade, --프로젝트 번호
	area VARCHAR2(50), --면적
	map VARCHAR2(50), --지도
	location VARCHAR2(20), --소재지
	scale VARCHAR2(10), --규모
	comple_year VARCHAR2(10), --준공년도
	quote VARCHAR2(20), --시세
	code VARHCHAR2(1) --담보분류
);

CREATE TABLE loan_list ( --<대출>
	loan_num NUMBER PRIMARY KEY --대출 번호
	loan_price VARCHAR2(20), --대출 금액
	user_num NUMBER REFERENCES user_list(user_num) on delete cascade, --유저 번호
	loan_period VARCHAR2(10), --대출기간
	loan_amount NUMBER(10), --대출금액
	loan_class VARCHAR2(10), --대출구분
	loan_purpose VARCHAR2(10), --대출목적
	repay_method VARCHAR2(10) --상환방식
	project_num NUMBER REFERENCES project_list(project_num) on delete cascade, --프로젝트 번호
	busi_num NUMBER REFERENCES busi_user_list(busi_num) on delete cascade --법인 유저 번호
);

CREATE TABLE invest_list ( --<투자>
	invest_num NUMBER PRIMARY KEY --투자 번호

);
